# Ottoman Conquest of Mamluks
country_event = {
	id = annexation.1
	title = "mamluk.title"
	desc = "mamluk.desc"
	picture = BATTLE_eventPicture
	
	fire_only_once = yes

	major = yes

	trigger = {
		NOT = { has_global_flag = disable_annexation }
        tag = TUR
		is_subject = no
		war_with = MAM
		is_neighbor_of = MAM
       		is_year = 1500
		NOT = {
			is_year = 1600
		}
        	adm_power = 300
           	controls = 361
		if = {
			limit = {
				MAM = {
					owns_core_province = 377
				}
			}
			controls = 377
		}
		if = {
			limit = {
				MAM = {
					owns_core_province = 382
				}
			}
			controls = 382
		}        
            	MAM = {
			ai = yes
			was_player = no
			is_subject = no
            		owns_core_province = 361
            		num_of_owned_provinces_with = {
				value = 7
				controlled_by = TUR
            		}
		}
		NOT = { has_country_flag = mamluk_accepted }
       		NOT = { has_country_flag = mamluk_refused }
	}

	mean_time_to_happen = {
		days = 30
	}

	option = {
		name = "mamluk.a" 
                set_country_flag = mamluk_accepted
                add_adm_power = -300
                add_dip_power = -200
                add_mil_power = -200
		MAM = {
                	every_owned_province = {
				limit = {
					AND = {
					is_core = MAM
                    			culture = al_misr_arabic
					}
				}
				cede_province = TUR
                add_province_modifier = {
					name = mamluks_rebels
					duration = 7300
		        }
				add_province_modifier = {
					name = mamluk_influence
					duration = 73000
		        }
				add_core = TUR
				add_local_autonomy = 100
		}
                	every_owned_province = {
				limit = {
					AND = {
						is_core = MAM
                				NOT = { culture = al_misr_arabic }
					}
				}
				cede_province = TUR
                		add_core = TUR
				add_local_autonomy = 75
			}
                	every_owned_province = {
				limit = {
					NOT = { is_core = MAM }
				}
				cede_province = TUR
				add_local_autonomy = 75
	      		}
		}
              	ai_chance = {
			factor = 80
		}
	}
	
	option	= {  
		name = "mamluk.b"
                set_country_flag = mamluk_refused
                add_stability_or_adm_power = yes
                ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				is_rival = MAM
			}
		}
	}
}

# Aq Qoyunlu Conquest of Qara Qoyunlu - Qara Qoyunlu Leader Dies

    id = annexation.6
	title = "qara_dies.title"
	desc = "qara_dies.desc"
	picture = BATTLE_eventPicture
	
	fire_only_once = yes

	major = yes

	trigger = {
		NOT = { has_global_flag = disable_annexation }
        tag = AKK
		is_subject = no
		war_with = QAR
		is_neighbor_of = QAR
        	is_year = 1450
		NOT = {
			is_year = 1550
		}    
        	QAR = {
			ai = yes
			was_player = no
			is_subject = no
		}
		NOT = { has_country_flag = qara_accepted }
        	NOT = { has_country_flag = qara_refused }
	}

	mean_time_to_happen = {
		months = 250
		modifier = {
			factor = 0.75
			has_heir = no
		}
		modifier = {
			factor = 0.75
			is_monarch_leader = yes
		}
		modifier = {
			factor = 0.75
			is_heir_leader = yes
		}
		modifier = {
			factor = 0.85
			NOT = { war_score = 0 }
		}
		modifier = {
			factor = 0.75
			NOT = { war_score = -25 }
		}
		modifier = {
			factor = 0.75
			NOT = { war_score = -50 }
		}
		modifier = {
			factor = 0.5
			NOT = { war_score = -75 }
		}
	}

	option = {
		name = "qara_dies.a" 
                set_country_flag = qara_accepted
		QAR = {
			if = {
				limit = {
					has_heir = yes
				}
				kill_heir = yes
			}
			tooltip = {
				kill_ruler = yes
			}
			hidden_effect = {
				define_ruler = {
					regency = yes
				}
			}
                	every_owned_province = {
				limit = {
					AND = {
					is_core = QAR
                    			culture = azerbaijani
					}
				}
				cede_province = AKK
                		add_core = AKK
				add_local_autonomy = 25
			}
                	every_owned_province = {
				limit = {
					AND = {
					is_core = QAR
                			NOT = { culture = azerbaijani }
					}
				}
				cede_province = AKK
                		add_core = AKK
				add_local_autonomy = 25
			}
                	every_owned_province = {
				limit = {
					NOT = { is_core = QAR }
				}
				cede_province = AKK
				add_local_autonomy = 25
			}
		}
	}
