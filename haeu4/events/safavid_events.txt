country_event = {
	id = safavid.1
	title = safavid.1.t
	desc = safavid.1.d
	picture = RELIGIOUS_WARS_eventPicture

	fire_only_once = yes

	trigger = {
		tag = ARL
		NOT = {
			has_ruler = "Joneid"
		}
	}

	mean_time_to_happen = {
		months = 24
	}

	option = {
		name = safavid.1.a
		ai_chance = {
			factor = 100
		}
		define_heir = {
			name = "Joneid"
			dynasty = "Safavi"
			adm = 3		
			dip = 6
			mil = 4
			age = 30
		}
		kill_ruler = yes
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = inspiring_leader_personality }
		}
		hidden_effect = {
			change_religion = shiite
			every_owned_province = {
				limit = {
					religion = sunni
				}
				change_religion = shiite
			}
		}
		custom_tooltip = safavid.1.a.tooltip
		define_ruler_to_general = {
			fire = 2
			shock = 3
			manuever = 3
			siege = 0
		}
		add_country_modifier = {
			name = rise_of_safavids
			duration = -1
		}
	}

	option = {
		name = safavid.1.b
		ai_chance = {
			factor = 0
		}
		capital_scope = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Joneid Safavi"
			}		
		}
	}
}
